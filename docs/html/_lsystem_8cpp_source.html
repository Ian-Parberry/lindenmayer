59 void LSystem::AddRule(const LProduction& rule){
60  if(rule.m_fProb < 1)m_bStochastic = true;
61 
62  auto p = m_mapRules.find(rule.m_chLHS);
63 
64  if(p == m_mapRules.end()){
65  std::vector<LProduction> v;
66  v.push_back(rule);
67  m_mapRules.insert(std::make_pair(rule.m_chLHS, v));
68  } //if
69 
70  else p->second.push_back(rule);
71 
72  //add rule to rule string for display
73 
74  m_strRuleString += rule.m_chLHS;
75  m_strRuleString += L" \u2192 " + (std::wstring)rule.m_strRHS;
76 
77  //a bit of fuss here to get the probability with only 2 digits precision
78 
79  if(m_bStochastic){
80  std::wstringstream wstream; //weird but necessary
81  wstream.precision(2); //2 digits after the decimal point
82  wstream << std::fixed << rule.m_fProb; //apply it
83  m_strRuleString += L" (" + wstream.str() + L")"; //append to rule string
84  } //if
85 
86  m_strRuleString += L"\n"; //end rule
87 } //AddRule

124 void LSystem::Generate(int n){
125  m_nGenerations = n;
126 
127  std::wstring* pSrc = &m_strBuffer[0]; //source buffer
128  std::wstring* pDest = &m_strBuffer[1]; //destination buffer
129 
130  *pSrc = m_strRoot; //copy root string to source buffer
131 
132  for(int i=0; i<n; i++){ //for each generation
133  pDest->clear();
134 
135  for(size_t i=0; i<pSrc->size(); i++){ //for each char in source
136  bool bRuleApplied = false; //whether a rule has been applied yet
137 
138  auto p = m_mapRules.find((*pSrc)[i]);
139 
140  if(p != m_mapRules.end()){
141  float fProb = 0; //cumulative probability
142  const float fRand = m_cRandom.randf(); //random value in [0, 1]
143 
144  for(auto rule: p->second){ //for each production that applies
145  fProb += rule.m_fProb; //accumulate probability
146 
147  if(fRand <= fProb){ //use the current rule
148  *pDest += rule.m_strRHS; //apply rule
149  bRuleApplied = true; //record that a rule was applied
150  break; //no need to try more rules
151  } //if
152  } //for
153  } //if
154 
155  if(!bRuleApplied) //no rule was applied to current symbol
156  *pDest += (*pSrc)[i]; //just copy over the current symbol
157  } //for
158 
159  std::swap(pSrc, pDest); //swap generation buffers
160  } //for
161 
162  m_pResult = pDest; //copy the latest string to the result string
163 } //Generate
