99  //dirty rectangle width and height
100 
101  const int nDirtyWidth = m_rectDirty.right - m_rectDirty.left;
102  const int nDirtyHt = m_rectDirty.bottom - m_rectDirty.top;
103 
104  //get client rectangle
105 
106  RECT rectClient; //for client rectangle
107  GetClientRect(m_hWnd, &rectClient); //get client rectangle
108  const int nClientWidth = rectClient.right - rectClient.left; //client width
109  const int nClientHt = rectClient.bottom - rectClient.top; //client height
110 
111  //fill client area with white background
112 
113  Gdiplus::SolidBrush brush(Gdiplus::Color::White);
114  graphics.FillRectangle(&brush, 0, 0, nClientWidth, nClientHt);
115 
116  //draw the dirty rectangle of the bitmap to the center of the client area
117 
118  const int x = max(0, (nClientWidth - nDirtyWidth)/2);
119  const int y = max(0, (nClientHt - nDirtyHt)/2);
120 
121  graphics.DrawImage(m_pBitmap, x, y, m_rectDirty.left, m_rectDirty.top,

149 void CMain::Draw(const TurtleDesc& d){
150  std::stack<StackFrame> stack; //stack frame
151 
152  //prepare to draw
153 
154  m_pGraphics->Clear(Gdiplus::Color::Transparent); //transparent background
155 
156  Gdiplus::PointF ptCurPos = d.m_ptStart; //current position
157  float angle = 0; //current orientation
158  float len = d.m_fLength; //current branch length
159 
160  Gdiplus::Pen pen(Gdiplus::Color::Black);
161  pen.SetWidth(d.m_fPointSize);
162 
163  //initialize the dirty rectangle to the start pixel
164 
165  m_rectDirty.left = int(std::floor(ptCurPos.X));
166  m_rectDirty.right = int(std::ceil (ptCurPos.X));
167  m_rectDirty.top = int(std::floor(ptCurPos.Y));
168  m_rectDirty.bottom = int(std::ceil (ptCurPos.Y));
169 
170  const std::wstring& s = m_cLSystem.GetString(); //shorthand for generated string
171 
172  //loop through characters of the string s, performing turtle graphics
173 
174  for(size_t i=0; i<s.size(); i++){
175  Gdiplus::PointF ptNextPos;
176 
177  switch(s[i]){
178  case 'L':
179  case 'R':
180  case 'F':
181  ptNextPos = ptCurPos - Gdiplus::PointF(-len*sinf(angle), len*cosf(angle));
182  m_pGraphics->DrawLine(&pen, ptCurPos, ptNextPos);
183  ptCurPos = ptNextPos;
184 
185  AddPointToRect(m_rectDirty, ptNextPos);
186  break;
187 
188  case '+': angle -= d.m_fAngleDelta; break;
189  case '-': angle += d.m_fAngleDelta; break;
190 
191  case '[':
192  stack.push(StackFrame(ptCurPos, angle, len));
193  len *= d.m_fLenMultiplier;
194  break;
195 
196  case ']': {
197  const StackFrame& sf = stack.top();
198 
199  ptCurPos = sf.m_ptPos;
200  angle = sf.m_fAngle;
201  len = sf.m_fLength;
202 
203  stack.pop(); //this must be last, obviously
204  } //case
205  break;
206  } //switch
207  } //for
208 
209  //make the dirty rectangle slightly larger to include lines on the edge
210 
211  const int delta = (int)std::ceil(d.m_fPointSize/2.0f); //amount to add
212  m_rectDirty.right += delta;
213  m_rectDirty.bottom += delta;
214 } //Draw

222 void CMain::Draw(){
223  TurtleDesc d; //turtle graphics descriptor
224  const float fDegToRad = (float)M_PI/180.0f; //degrees to radians
225 
226  switch(m_eLineThickness){
227  case LineThickness::Thin: d.m_fPointSize = 1.0f; break;
228  case LineThickness::Thick: d.m_fPointSize = 2.0f; break;
229 
230  default: d.m_fPointSize = 1.0f; break;
231  } //switch
232 
233  const float midx = 0.5f*m_pBitmap->GetWidth();
234  const float midy = 0.5f*m_pBitmap->GetHeight();
235  const float bottomy = m_pBitmap->GetHeight() - 64.0f;
236 
237  const Gdiplus::PointF ptBottomCenter(midx, bottomy);
238  const Gdiplus::PointF ptMidCenter(midx, midy);
239 
240  switch(m_nType){ //the angle deltas are cribbed from ABOP
241  case IDM_LSYS_PLANT_A:
242  d = TurtleDesc(ptBottomCenter, fDegToRad*22.7f, 8.0f);
243  break;
244 
245  case IDM_LSYS_PLANT_B:
246  d = TurtleDesc(ptBottomCenter, fDegToRad*20.0f, 20.0f);
247  break;
248 
249  case IDM_LSYS_PLANT_C:
250  d = TurtleDesc(ptBottomCenter, fDegToRad*22.5f, 12.0f);
251  break;
252 
253  case IDM_LSYS_PLANT_D:
254  d = TurtleDesc(ptBottomCenter, fDegToRad*20.0f, 5.0f);
255  break;
256 
257  case IDM_LSYS_PLANT_E:
258  d = TurtleDesc(ptBottomCenter, fDegToRad*25.7f, 5.0f);
259  break;
260 
261  case IDM_LSYS_PLANT_F:
262  d = TurtleDesc(ptBottomCenter, fDegToRad*22.5f, 16.0f);
263  break;
264 
265  case IDM_LSYS_BRANCHING:
266  d = TurtleDesc(ptBottomCenter, 0.37f, 8.0f);
267  break;
268 
269  case IDM_LSYS_HEXGOSPER:
270  d = TurtleDesc(ptMidCenter, float(M_PI/3), 12.0f);
271  break;
272  } //switch
273 
274  // line thickness
275 
276  switch(m_eLineThickness){
277  case LineThickness::Thin: d.m_fPointSize = 1.0f; break;
278  case LineThickness::Thick: d.m_fPointSize = 2.0f; break;
279 
280  default: d.m_fPointSize = 1.0f; break;
281  } //switch
282 
283  Draw(d);
284  InvalidateRect(m_hWnd, nullptr, TRUE);
285 } //Draw

383 void CMain::SetRules(){
384  m_cLSystem.Clear();
385 
386  switch(m_nType){
387  case IDM_LSYS_BRANCHING:
388  m_cLSystem.SetRoot(L"F");
389  m_cLSystem.AddRule(LProduction(L'F', L"F[+F]F[-F]F", 0.33f));
390  m_cLSystem.AddRule(LProduction(L'F', L"F[+F]F", 0.33f));
391  m_cLSystem.AddRule(LProduction(L'F', L"F[-F]F", 0.34f));
392  break;
393 
394  case IDM_LSYS_PLANT_A:
395  m_cLSystem.SetRoot(L"F");
396  m_cLSystem.AddRule(LProduction(L'F', L"F[+F]F[-F]F"));
397  break;
398 
399  case IDM_LSYS_PLANT_B:
400  m_cLSystem.SetRoot(L"F");
401  m_cLSystem.AddRule(LProduction(L'F', L"F[+F]F[-F][F]"));
402  break;
403 
404  case IDM_LSYS_PLANT_C:
405  m_cLSystem.SetRoot(L"F");
406  m_cLSystem.AddRule(LProduction(L'F', L"FF-[-F+F+F]+[+F-F-F]"));
407  break;
408 
409  case IDM_LSYS_PLANT_D:
410  m_cLSystem.SetRoot(L"X");
411  m_cLSystem.AddRule(LProduction(L'X', L"F[+X]F[-X]+X"));
412  m_cLSystem.AddRule(LProduction(L'F', L"FF"));
413  break;
414 
415  case IDM_LSYS_PLANT_E:
416  m_cLSystem.SetRoot(L"X");
417  m_cLSystem.AddRule(LProduction(L'X', L"F[+X][-X]FX"));
418  m_cLSystem.AddRule(LProduction(L'F', L"FF"));
419  break;
420 
421  case IDM_LSYS_PLANT_F:
422  m_cLSystem.SetRoot(L"X");
423  m_cLSystem.AddRule(LProduction(L'X', L"F-[ [X]+X]+F[+FX]-X"));
424  m_cLSystem.AddRule(LProduction(L'F', L"FF"));
425  break;
426 
427  case IDM_LSYS_HEXGOSPER:
428  m_cLSystem.SetRoot(L"L");
429  m_cLSystem.AddRule(LProduction(L'L', L"L+R++R-L--LL-R+"));
430  m_cLSystem.AddRule(LProduction(L'R', L"-L+RR++R+L--L-R"));
431  break;
432  } //switch
433 } //SetRules
