102 void CMain::OnPaint(){
103  PAINTSTRUCT ps; //paint structure
104  HDC hdc = BeginPaint(m_hWnd, &ps); //device context
105  Gdiplus::Graphics graphics(hdc); //GDI+ graphics object
106 
107  //dirty rectangle width and height
108 
109  const int nDirtyWidth = m_rectDirty.right - m_rectDirty.left;
110  const int nDirtyHt = m_rectDirty.bottom - m_rectDirty.top;
111 
112  //get client rectangle
113 
114  RECT rectClient; //for client rectangle
115  GetClientRect(m_hWnd, &rectClient); //get client rectangle
116  const int nClientWidth = rectClient.right - rectClient.left; //client width
117  const int nClientHt = rectClient.bottom - rectClient.top; //client height
118 
119  //fill client area with white background
120 
121  graphics.Clear(Gdiplus::Color::White); //white background
122 
123  //draw the dirty rectangle of the bitmap to the center of the client area
124 
125  int textwidth = m_bShowRules? (int)std::ceil(GetRuleStrWidth(graphics)): 0;
126 
127  const int margin = 10; //default margin
128 
129  const float xscale = float(nClientWidth
130  - (textwidth > 0? 3: 2)*margin - textwidth)/nDirtyWidth;
131  const float yscale = float(nClientHt - 2*margin)/nDirtyHt;
132  const float scale = min(min(xscale, yscale), 1);
133 
134  Gdiplus::Rect rectDest;
135 
136  rectDest.Width = (int)std::floor(scale*nDirtyWidth);
137  rectDest.Height = (int)std::floor(scale*nDirtyHt);
138 
139  rectDest.X = max(2*margin + textwidth,
140  (nClientWidth - rectDest.Width + textwidth)/2);
141  rectDest.Y = max(margin, (nClientHt - rectDest.Height)/2);
142 
143  graphics.DrawImage(m_pBitmap, rectDest, m_rectDirty.left, m_rectDirty.top,
144  nDirtyWidth, nDirtyHt, Gdiplus::UnitPixel);
145 
146  //draw the rules on the screen (note: NOT on the bitmap)
147 
148  if(m_bShowRules)
149  DrawRules(graphics, Gdiplus::PointF(margin, margin));
150 
151  EndPaint(m_hWnd, &ps); //this must be done last
152 } //OnPaint

172 void CMain::Draw(const TurtleDesc& d){
173  std::stack<StackFrame> stack; //stack frame
174 
175  //prepare to draw
176 
177  m_pGraphics->Clear(Gdiplus::Color::Transparent); //transparent background
178 
179  Gdiplus::PointF ptCurPos = d.m_ptStart; //current position
180  float angle = 0; //current orientation
181  float len = d.m_fLength; //current branch length
182 
183  Gdiplus::Pen pen(Gdiplus::Color::Black);
184  pen.SetWidth(d.m_fPointSize);
185 
186  //initialize the dirty rectangle to the start pixel
187 
188  m_rectDirty.left = int(std::floor(ptCurPos.X));
189  m_rectDirty.right = int(std::ceil (ptCurPos.X));
190  m_rectDirty.top = int(std::floor(ptCurPos.Y));
191  m_rectDirty.bottom = int(std::ceil (ptCurPos.Y));
192 
193  const std::wstring& s = m_cLSystem.GetString(); //shorthand for generated string
194 
195  //loop through characters of the string s, performing turtle graphics
196 
197  for(size_t i=0; i<s.size(); i++){
198  Gdiplus::PointF ptNextPos;
199 
200  switch(s[i]){
201  case 'L':
202  case 'R':
203  case 'F':
204  ptNextPos = ptCurPos - Gdiplus::PointF(-len*sinf(angle), len*cosf(angle));
205  m_pGraphics->DrawLine(&pen, ptCurPos, ptNextPos);
206  ptCurPos = ptNextPos;
207 
208  AddPointToRect(m_rectDirty, ptNextPos);
209  break;
210 
211  case '+': angle -= d.m_fAngleDelta; break;
212  case '-': angle += d.m_fAngleDelta; break;
213 
214  case '[':
215  stack.push(StackFrame(ptCurPos, angle, len));
216  len *= d.m_fLenMultiplier;
217  break;
218 
219  case ']': {
220  const StackFrame& sf = stack.top();
221 
222  ptCurPos = sf.m_ptPos;
223  angle = sf.m_fAngle;
224  len = sf.m_fLength;
225 
226  stack.pop(); //this must be last, obviously
227  } //case
228  break;
229  } //switch
230  } //for
231 
232  //make the dirty rectangle slightly larger to include lines on the edge
233 
234  const int delta = (int)std::ceil(d.m_fPointSize/2.0f); //amount to add
235  m_rectDirty.right += delta;
236  m_rectDirty.bottom += delta;
237 } //Draw

246 void CMain::Draw(){

249 
250  const float midx = 0.5f*m_pBitmap->GetWidth();

253 
254  const Gdiplus::PointF ptBottomCenter(midx, bottomy);
255  const Gdiplus::PointF ptMidCenter(midx, midy);
256 
257  switch(m_nType){ //the angle deltas are cribbed from ABOP

260  break;
261

264  break;
265

267  d = TurtleDesc(ptBottomCenter, fDegToRad*22.5f, 12.0f);
268  break;
269 
270  case IDM_LSYS_PLANT_D:
271  d = TurtleDesc(ptBottomCenter, fDegToRad*20.0f, 5.0f);
272  break;
273 
274  case IDM_LSYS_PLANT_E:
275  d = TurtleDesc(ptBottomCenter, fDegToRad*25.7f, 5.0f);
276  break;
277 
278  case IDM_LSYS_PLANT_F:
279  d = TurtleDesc(ptBottomCenter, fDegToRad*22.5f, 16.0f);

389 void CMain::SetRules(){
390  m_cLSystem.Clear();
391 
392  switch(m_nType){

394  m_cLSystem.SetRoot(L"F");
395  m_cLSystem.AddRule(LProduction(L'F', L"F[+F]F[-F]F", 0.33f));
396  m_cLSystem.AddRule(LProduction(L'F', L"F[+F]F", 0.33f));
397  m_cLSystem.AddRule(LProduction(L'F', L"F[-F]F", 0.34f));
398  break;
399 
400  case IDM_LSYS_PLANT_A:
401  m_cLSystem.SetRoot(L"F");
402  m_cLSystem.AddRule(LProduction(L'F', L"F[+F]F[-F]F"));
403  break;
404 
405  case IDM_LSYS_PLANT_B:
406  m_cLSystem.SetRoot(L"F");
407  m_cLSystem.AddRule(LProduction(L'F', L"F[+F]F[-F][F]"));
408  break;

410  case IDM_LSYS_PLANT_C:
411  m_cLSystem.SetRoot(L"F");
412  m_cLSystem.AddRule(LProduction(L'F', L"FF-[-F+F+F]+[+F-F-F]"));
413  break;
414 
415  case IDM_LSYS_PLANT_D:
416  m_cLSystem.SetRoot(L"X");
417  m_cLSystem.AddRule(LProduction(L'X', L"F[+X]F[-X]+X"));
418  m_cLSystem.AddRule(LProduction(L'F', L"FF"));

424  m_cLSystem.AddRule(LProduction(L'F', L"FF"));
425  break;
426 
427  case IDM_LSYS_PLANT_F:
428  m_cLSystem.SetRoot(L"X");
429  m_cLSystem.AddRule(LProduction(L'X', L"F-[ [X]+X]+F[+FX]-X"));
430  m_cLSystem.AddRule(LProduction(L'F', L"FF"));
431  break;
432 
433  case IDM_LSYS_HEXGOSPER:
434  m_cLSystem.SetRoot(L"L");
435  m_cLSystem.AddRule(LProduction(L'L', L"L+R++R-L--LL-R+"));

438  } //switch

554 int CMain::GetRuleStrWidth(Gdiplus::Graphics& graphics){
555  const std::wstring& strRules = m_cLSystem.GetRuleString(); //shorthand
556  const Gdiplus::RectF r = GetClientRectF(m_hWnd); //client rectangle
557 
558  const Gdiplus::CharacterRange charRange(0, (int)strRules.size());
559  Gdiplus::StringFormat sf;
560  sf.SetMeasurableCharacterRanges(1, &charRange);
561 
562  const int n = sf.GetMeasurableCharacterRangeCount();
563  Gdiplus::Region* rg = new Gdiplus::Region[n]; //array of regions
564  graphics.MeasureCharacterRanges(strRules.c_str(), -1, m_pFont, r, &sf, n, rg);
565 
566  //find the maximum region width in pixels
567 
568  int width = 0; //return result
569 
570  for(int i=0; i<n; i++){ //for each region
571  Gdiplus::Rect r; //for bounding rectangle of current region
572  rg[i].GetBounds(&r, &graphics); //get bounding rectangle of current region
573  width = max(width, r.Width); //see if current region width is largest so far
574  } //for
575 
576  //cleanup and exit
577 
578  delete [] rg;
579  return width;
580 } //GetRuleStrWidth
